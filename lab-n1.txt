#include <stdio.h>

int main() {
    // Вхідні дані згідно з варіантом
    int a = 1;
    int b = 2;
    int c = 3;
    
    int V, S;

    // Обчислення об'єму: V = a * b * c
    V = a * b * c;

    // Обчислення площі поверхні: S = 2 * (ab + bc + ac)
    S = 2 * (a * b + b * c + a * c);

    // Виведення результатів
    printf("Вхідні дані: a = %d, b = %d, c = %d\n", a, b, c);
    printf("Вихідні дані: V = %d, S = %d\n", V, S);

    return 0;
}
